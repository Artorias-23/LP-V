}

// POWER OF 2
public double powerOf2(double num) throws RemoteException{
    return Math.pow(num, 2);
}

// STRING COMPARISON (RETURNS LONGER STRING)
public String longerString(String s1, String s2) throws RemoteException{
    return (s1.length() >= s2.length()) ? s1 : s2;
}

// CELSIUS TO FAHRENHEIT
public double celsiusToFahrenheit(double c) throws RemoteException{
    return (c * 9 / 5) + 32;
}

// MILES TO KILOMETERS
public double milesToKm(double miles) throws RemoteException{
    return miles * 1.60934;
}

// FACTORIAL
public long factorial(int n) throws RemoteException{
    long fact = 1;
    for(int i = 1; i <= n; i++){
        fact *= i;
    }
    return fact;
}

// COUNT VOWELS
public int countVowels(String str) throws RemoteException{
    int count = 0;
    str = str.toLowerCase();

    for(int i = 0; i < str.length(); i++){
        char ch = str.charAt(i);

        if(ch == 'a' || ch == 'e' || ch == 'i' || 
           ch == 'o' || ch == 'u'){
            count++;
        }
    }

    return count;
}
